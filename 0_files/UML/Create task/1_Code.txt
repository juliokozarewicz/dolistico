title Sequence Diagram - POST /api/v1/tasks

actor User

User -> NGINX : POST /api/v1/tasks\nAuthorization: Bearer <jwt>
NGINX -> TasksCreateController : forward request

TasksCreateController -> TasksCreateUseCase : execute(input)

TasksCreateUseCase -> TasksCreateInput : new input from DTO
TasksCreateUseCase -> TasksEntity : validateBusinessRules(input)

alt Invalid business rule
    TasksEntity --> TasksCreateUseCase : throw DomainException
    TasksCreateUseCase --> TasksCreateController : 400 Bad Request
    TasksCreateController --> User : error response
    return
end

TasksCreateUseCase -> TasksModel : convert Entity → Model

TasksCreateUseCase -> TasksEventProducer : publish(task.created)
TasksEventProducer -> Kafka : send message

Kafka -> TasksEventConsumer : deliver event

TasksEventConsumer -> TasksModel : toModel(event)
TasksEventConsumer -> TasksRepositoryImpl : save(model)
TasksRepositoryImpl -> Database : INSERT task

TasksCreateUseCase --> TasksCreateController : StandardResponseDTO
TasksCreateController --> User : 201 Created

title Activity Diagram - Create Task (Business Rules)

start

partition "Controller" {
    :Receive TasksCreateInput;
}

-->

partition "UseCase" {
    :execute(input);
    :Map input → TasksEntity;
}

-->

partition "Domain (Business Rules)" {

    :validateBusinessRules();

    if (Missing required fields?) then (yes)
        :throw DomainException(REQUIRED_FIELD);
        stop
    endif

    if (taskName length < 3?) then (yes)
        :throw DomainException(INVALID_NAME);
        stop
    endif

    if (priority < 1 or > 5?) then (yes)
        :throw DomainException(INVALID_PRIORITY);
        stop
    endif

    if (endTime < startTime?) then (yes)
        :throw DomainException(INVALID_DATES);
        stop
    endif

    if (dueDate < today?) then (yes)
        :throw DomainException(INVALID_DATES);
        stop
    endif

    if (allDay = true AND (startTime or endTime not null)?) then (yes)
        :throw DomainException(INVALID_ALLDAY);
        stop
    endif

    if (reminderTime after dueDate or endTime?) then (yes)
        :throw DomainException(INVALID_REMINDER);
        stop
    endif

    if (Task exists same name + same date?) then (yes)
        :throw DomainException(DUPLICATED_TASK);
        stop
    endif

    :All rules approved;
}

-->

partition "Entity + Producer" {
    :TasksEntity.create();
    :Publish task.created event;
}

-->

partition "Messaging + Consumer" {
    :Kafka stores event;
    :TasksEventConsumer receives event;
    :Convert → TasksModel;
    :TasksRepositoryImpl.save(model);
}

stop
